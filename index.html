import streamlit as st import geopandas as gpd import folium from streamlit_folium import st_folium # Load GeoJSON data of Alabama gdf = gpd.read_file('alabama_counties.geojson') # Streamlit app layout st.title("Alabama GIS App") # Dropdown menu to select a county selected_county = st.selectbox("Select a County", gdf['NAME'].unique()) # Filter the GeoDataFrame for the selected county selected_gdf = gdf[gdf['NAME'] == selected_county] # Create a Folium map centered around Alabama m = folium.Map(location=[32.806671, -86.791130], zoom_start=7) # Add selected county to the map folium.GeoJson(selected_gdf).add_to(m) # Display the map in Streamlit st_folium(m, width=700, height=500)
